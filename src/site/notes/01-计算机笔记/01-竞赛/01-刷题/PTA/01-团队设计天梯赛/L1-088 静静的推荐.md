# 代码实现
```c++
#include <bits/stdc++.h>
using namespace std;

#define x first
#define y second

typedef pair<int, int> pii;

const int N = 1e5 + 10;
int n, k, s, cnt;
pii a[N];

int main()
{
    scanf("%d%d%d", &n, &k, &s);
    for (int i = 0; i < n; i++) scanf("%d%d", &a[i].x, &a[i].y);
    
    sort(a, a + n);


    vector<int> score(300, k - 1);
    for (int i = 0; i < n; i++) {
        if (a[i].x < 175) continue;
        cnt++;
        int j = i + 1;
        while (j < n && a[j].x == a[i].x) {
            if (a[j].y >= s) {
                cnt++;
            } else if (score[a[i].x]) {
                score[a[i].x]--;
                cnt++;
            }
            j++;
        }
        i = j - 1;
    }
    printf("%d", cnt);
    return 0;
}
```
