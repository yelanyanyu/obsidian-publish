可以想象斗地主的情形。当我们拿到了一张比较小的牌是，我们会将这个牌插入到离这张牌最近的比它小的数右边。
# 实现
以下是插入升序排序的实现方式。
```c++
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1e5 + 10;
int n;
int a[N];

void insert_sort() {
    for (int i = 1; i < n; i++) {
        int t = a[i], j = i;
        while (j && a[j - 1] > t) {
            a[j] = a[j - 1];
            j--;
        }
        a[j] = t;
    }
}

int main()
{
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    
    insert_sort();
    for (int i = 0; i < n; i++) cout << a[i] << " ";
    return 0;
}
```
